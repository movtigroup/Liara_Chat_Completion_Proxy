Usage and Code Samples

HTTP (POST) Request Sample

POST /api/v1/chat/completions
Authorization: Bearer YOUR_API_KEY

{
  "model": "openai/gpt-4o-mini",
  "messages": [
    {"role": "user", "content": "Hello, how are you?"} 
  ],
  "temperature": 0.7,
  "max_tokens": 500
}

WebSocket (JavaScript) Sample

const ws = new WebSocket('ws://YOUR_DOMAIN/ws/v1/chat/completions');

ws.onopen = () => {
  // First, send API Key for authentication
  ws.send(JSON.stringify({
    api_key: "Bearer YOUR_API_KEY" // Ensure Bearer prefix
  }));
  
  // Then, send chat configuration
  ws.send(JSON.stringify({
    model: "openai/gpt-4o-mini",
    messages: [{role: "user", "content": "Hello"}], // Translated
    stream: true // stream is implicitly true for WebSockets with this proxy
  }));
};

ws.onmessage = (event) => {
  const data = JSON.parse(event.data);
  console.log(data);
  // Handle streamed chunks or final error messages
};

ws.onerror = (error) => {
  console.error("WebSocket Error:", error);
};

ws.onclose = (event) => {
  console.log("WebSocket closed:", event.code, event.reason);
};
